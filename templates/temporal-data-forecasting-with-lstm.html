def dataresize(data, n_prior_steps, n_next_steps, whichset, m_train, m_val, m_test, continuous_way):
      x = list()
      y = list()
      if continuous_way:
          if whichset == "Training":
              start, end = n_prior_steps, m_train - n_next_steps
          elif whichset == "Validation":
              start, end = m_train, m_train + m_val - n_next_steps
          elif whichset == "Test":
              start, end = m_train + m_val, m_train + m_val + m_test - n_next_steps
      else:
          if whichset == "Training":
              start, end = n_prior_steps, m_train - n_next_steps
          elif whichset == "Validation":
              start, end = m_train + n_prior_steps, m_train + m_val - n_next_steps
          elif whichset == "Test":
              start, end = m_train + m_val + n_prior_steps, m_train + m_val + m_test - n_next_steps
      for i in range(start, end+1):
          sample_x = data[i-n_prior_steps:i]
          sample_y = data[i:i+n_next_steps]
          x.append(sample_x)
          y.append(sample_y)
      # Tranform the data set features and labels into arrays
      x = np.array(x)
      y = np.array(y).reshape((len(y), sample_y.shape[0]))
      return x, y

  def lossplot(history, **kwargs):
      title = kwargs.get('title', None)
      loss_name = kwargs.get('loss_name', None)
      metrics_name = kwargs.get('metrics_name', None)
      loss = history.history["loss"]
      val_loss = history.history["val_loss"]
      metrics_values = history.history[metrics_name]
      val_metrics_values = history.history["val_"+metrics_name]
      epochs = range(len(loss))
      
      plt.figure(figsize=(15,5))
      plt.subplot(121)
      plt.plot(epochs, loss, "b", label="Training "+loss_name)
      plt.plot(epochs, val_loss, "green", label="Validation "+loss_name)
      plt.title("Training and Validation {}: {}".format(loss_name,title))
      plt.xlabel("Epochs")
      plt.ylabel(loss_name)
      plt.legend(frameon=False)
      plt.subplot(122)
      plt.plot(epochs, metrics_values, "orange", label="Training "+metrics_name)
      plt.plot(epochs, val_metrics_values, "r", label="Validation "+metrics_name)
      plt.title("Training and Validation {}: {}".format(metrics_name, title))
      plt.xlabel("Epochs")
      plt.ylabel(metrics_name)
      plt.legend(frameon=False)
      plt.show()
